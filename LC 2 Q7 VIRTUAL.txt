#include<iostream>
using namespace std;
class student
{
    protected:
       int roll_no;
    public:
       void get_number(int);
       void put_number(void);
};
void student::get_number(int a)
{
    roll_no=a;
}
void student::put_number()
{
    cout<<"Roll no : "<<roll_no<<endl;
}

};

    sub2=y;
}

{
