class matrix
{
   int a[s][s],x,y;
    public:
   void get();
   void put();
   matrix operator+(matrix);
   matrix operator-(matrix);
   matrix operator*(matrix);
   matrix transpose();
};
void matrix::get()
{
   cout<<"Enter the order of Matrix "<<" :\n";
   cin>>x>>y;
   cout<<"Enter the Matrix "<<" :\n";

         cin>>a[i][j];
 
}
void matrix::put()
{
   cout<<"The Ans is:\n";
   for(int i=0;i<x;i++)
   {
      cout<<"\n\t";
      for(int j=0;j<y;j++)
         cout<<a[i][j]<<" ";
   }
}
matrix matrix::operator+(matrix b)
{
   matrix r;
   if((x!=b.x)||(y!=b.y))
   {
      cout<<"\n\tMatrix Addition is not possible the result is incorrect\n\n";
      r.x=0;
      r.y=0;
   }
   else
   {
      r.x=x;
      r.y=y;
   }
   for(int i=0;i<x;i++)
      for(int j=0;j<y;j++)
         r.a[i][j]=a[i][j]+b.a[i][j];
   return r;
}

matrix matrix::operator*(matrix b)
{
   matrix r;
   if((x!=b.y)||(y!=b.x))
   {
      cout<<"Matrix Multiplication is not possible";
      r.x=0;
      r.y=0;
   }
   else
   {
      r.x=x;
      r.y=b.y;
   }
   for(int i=0;i<s;i++)
      for(int j=0;j<s;j++)
         r.a[i][j]=0;
   for(int i=0;i<x;i++)
      for(int j=0;j<b.y;j++)
         for(int k=0;(k<y)||(k<b.x);k++)
            r.a[i][j]+=a[i][k]*b.a[k][j];
 
   return r;
}
matrix matrix::transpose()
{
   matrix r;
   for(int i=0;i<x;i++)
      for(int j=0;j<y;j++)
         r.a[i][j]=a[j][i];
   r.x=x;
   r.y=y;
   return r;
}
